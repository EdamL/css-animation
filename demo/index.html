CSS animations Plugin Demo

cssSlideToggle:

var form = $('form[name="slide-demo"]');
var animationElement = $('#slide-element');

form.find('button').on('click', function() {
   
   var duration = form.find('input[name="duration"]').val();
   var easing = form.find('input[name="easing"]').val();
   
   animationElement.html('');
   
   animationElement.cssSlideToggle(duration, easing, function() {
      animationElement.html('Done!');
   });
});

cssAnimate:

var form = $('form[name="animate-demo"]');
var animationElement = $('#animate-element');

form.find('button').on('click', function() {
   
   var duration = form.find('input[name="duration"]').val();
   var width = form.find('input[name="width"]').val();
   var height = form.find('input[name="height"]').val();
   var opacity = form.find('input[name="opacity"]').val();
   var rotation = form.find('input[name="rotation"]').val();
   var easing = form.find('input[name="easing"]').val();
   
   animationElement.html('');
   
   console.log(width, height, opacity, rotation)
   
   animationElement.cssAnimate({
      'width' : width + 'px',
      'height' : height + 'px',
      'opacity' : opacity,
      'transform' : 'rotate(' + rotation + 'deg)'
   }, duration, easing, function() {
      animationElement.html('Done!');
   });
});
